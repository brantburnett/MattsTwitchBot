{
  "SoundEffects": [
    {
      "SoundEffectName": "!rimshot"
    },
    {
      "SoundEffectName": "!badumtss"
    },
    {
      "SoundEffectName": "!laugh"
    },
    {
      "SoundEffectName": "!sadtrombone"
    }
  ]
}
